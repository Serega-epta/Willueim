#Слишком мало, слишком поздно.
country_event = {
	id = Unification_event.4
	title = Unification_event.4.t
	desc = Unification_event.4.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		name = Unification_event.4.a
		custom_effect_tooltip = Unification_deadline_fail_tt
		IF = {
			limit = {
				filtration_station > 0 
				filtration_station < 4
			}
			add_to_variable = {
    			var = monthly_population_var
    			value = -0.4
			}
			add_manpower = -400000
		}
		else_if = {
			limit = {
				filtration_station > 3 
				filtration_station < 6
			}
			add_to_variable = {
    			var = monthly_population_var
    			value = -0.3
			}
			add_manpower = -300000
		}
		else_if = {
			limit = {
				filtration_station > 6 
				filtration_station < 8
			}
			add_to_variable = {
    			var = monthly_population_var
    			value = -0.2
			}
			add_manpower = -200000
		}
		else = {
			add_to_variable = {
    			var = monthly_population_var
    			value = -0.1
			}
			add_manpower = -100000
		}
	}
}
