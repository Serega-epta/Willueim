#Завершение Первой фазы
country_event = {
	id = Sibir_events.1
	title = Sibir_events.1.t
	desc = Sibir_events.1.d
	trigger = {
        check_variable = {
            vaccine_progress_var > 99
        }
		has_country_flag = vaccine_phase_1_flag
    }
	option = {
		name = Sibir_events.1.a
		clr_country_flag = vaccine_phase_1_flag
        set_country_flag = vaccine_phase_2_flag
        set_variable = {
    		var = vaccine_progress_var
    		value = 0
		}
		set_variable = {
    		var = monthly_vaccine_progress_var
    		value = 0
		}
		remove_decision = Build_new_hospitals
		remove_decision = Increase_investment_in_medicine
		remove_decision = Expand_medical_faculties
	}
}
