on_actions = {
	on_startup = {	
		effect = {
			every_country = {
				if = {
					limit = { is_ai = no }
					set_country_flag = opening_screen_show
				}
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = { original_tag = ODS }
				add_dynamic_modifier = { ODS_new_land }
			}
		}
	}
	on_nuke_drop = {
		effect = {
			FROM = {
				goto_state = {
					add_manpower = -100000
				}
			}
		}
	}
}
