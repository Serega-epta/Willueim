Smolensk_war_finished = {
	AND = {
		NAR = { 
			OR = { 
				exists = no 
				AND = {
					has_war = no 
					owns_state = 242
					owns_state = 210
					owns_state = 207
					owns_state = 931					
				} 
			}
		} 
		SMK = { 
			OR = { 
				exists = no
				AND = {
					has_war = no 
					owns_state = 246
					owns_state = 263					
				} 
			}
		} 
	}
}

	}
}
