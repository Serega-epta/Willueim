scripted_gui = {
    economy_view = {
        context_type = player_context
	parent_window_token = politics_tab
	window_name = "economy_view"
	dirty = country_economy
	
	visible = {
		is_ai = no
	}
		
	effects = {
		economy_button_click = {
			if = {
				limit = { NOT = { has_variable = country_economy } }
				set_variable = { country_economy = 1 }
			}
			else = {
				clear_variable = country_economy
			}
			update_position = yes
		}
	}
    }

    economy_main = {
        context_type = player_context	
	parent_window_token = politics_tab
	window_name = "economy_main"
		
	visible = {
            	check_variable = { country_economy > 0 }
	}
		
	effects = {
		left_1_click = {
			ROOT={
				military_production = -1
				subtract_from_variable = { FROM.afactoriesVariable = 1 }
				subtract_from_variable = { FROM.armyVariable = 10 }
				subtract_from_variable = { FROM.economyVariable = 100 }
			}
		}
		right_1_click = {
			ROOT={
				military_production = 1
				add_to_variable = { FROM.afactoriesVariable = 1 }
				add_to_variable = { FROM.armyVariable = 10 }
				add_to_variable = { FROM.economyVariable = 100 }
			}
		}
		left_2_click = {
			ROOT={
				general_production = -1
				subtract_from_variable = { FROM.cfactoriesVariable = 1 }
				subtract_from_variable = { FROM.civilianVariable = 10 }
				subtract_from_variable = { FROM.economyVariable = 100 }
			}
		}
		right_2_click = {
			ROOT={
				general_production = 1
				add_to_variable = { FROM.cfactoriesVariable = 1 }
				add_to_variable = { FROM.civilianVariable = 10 }
				add_to_variable = { FROM.economyVariable = 100 }
			}
		}
	trigger = {
		left_1_click_enabled = {	
			ROOT={
				
				check_variable = { FROM.economyVariable > 0 }
			}
		}
		left_2_click_enabled = {	
			ROOT={
				
				check_variable = { FROM.economyVariable > 0 }
			}
		}
	}	
	}
    }
}
