scripted_gui = {
    economy_view = {
        context_type = player_context
	parent_window_token = politics_tab
	window_name = "economy_view"
	dirty = country_economy
	
	visible = {
	}
		
	effects = {
		economy_button_click = {
			if = {
				limit = { NOT = { has_variable = country_economy } }
				set_variable = { country_economy = 1 }
			}
			else = {
				clear_variable = country_economy
			}
		}
	}
    }

    economy_main = {
        context_type = player_context	
	parent_window_token = politics_tab
	window_name = "economy_main"
		
	visible = {
            	check_variable = { country_economy > 0 }
	}
		
	effects = {
		left_1_click = {
			if = {
				limit = { OR = { has_political_power > 25 has_political_power = 25 } }
				add_political_power = -25
				add_to_variable = { economyVariable = 100 }
				subtract_from_variable = { asponsorVariable = 1 }
			}
			else = {
				limit = { OR = { has_political_power < 0 has_political_power = 0 } }
				add_political_power = 0
				add_to_variable = { economyVariable = 0 }
				subtract_from_variable = { asponsorVariable = 0 }
			}
		}
		left_2_click = {
			if = {
				limit = { OR = { has_political_power > 25 has_political_power = 25 } }
				add_political_power = -25
				add_to_variable = { economyVariable = 100 }
				subtract_from_variable = { csponsorVariable = 1 }
			}
			else = {
				limit = { OR = { has_political_power < 0 has_political_power = 0 } }
				add_political_power = 0
				add_to_variable = { economyVariable = 0 }
				subtract_from_variable = { asponsorVariable = 0 }
			}
		}
		right_1_click = {
			if = {
				limit = { OR = { check_variable = { economyVariable > 100 } check_variable = { economyVariable = 100 } } OR = { has_political_power > 25 has_political_power = 25 } }
				add_political_power = -25
				subtract_from_variable = { economyVariable = 100 }
				add_to_variable = { asponsorVariable = 1 }
			}
			else = {
				limit = { check_variable = { economyVariable < 100 } OR = { has_political_power < 0 has_political_power = 0 } }
				add_political_power = 0
				subtract_from_variable = { economyVariable = 0 }
				add_to_variable = { asponsorVariable = 0 }
			}
		}
		right_2_click = {
			if = {
				limit = { OR = { check_variable = { economyVariable > 100 } check_variable = { economyVariable = 100 } } OR = { has_political_power > 25 has_political_power = 25 } }
				add_political_power = -25
				subtract_from_variable = { economyVariable = 100 }
				add_to_variable = { csponsorVariable = 1 }
			}
			else = {
				limit = { check_variable = { economyVariable < 100 } OR = { has_political_power < 0 has_political_power = 0 } }
				add_political_power = 0
				subtract_from_variable = { economyVariable = 0 }
				add_to_variable = { asponsorVariable = 0 }
			}
		}
	}

	triggers = {
		
	}	
    }
}
