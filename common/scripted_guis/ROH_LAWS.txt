			AND = {
				check_variable = { okno_zakonow_priziva_opend = 1 }

			AND = {
				check_variable = { okno_zakonow_priziva_opend = 1 }
				has_idea = admiral_recruting
			}
		}			
	}		
	effects = {
		free_recruting_click = {

					add_to_variable = { Chernii_rubli = -5000 }
				}	
		}
		limited_recruting_click = {

				}
		}
