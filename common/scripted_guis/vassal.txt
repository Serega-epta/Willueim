scripted_gui = { 
    vassal = {
        context_type = selected_country_context
		parent_window_token = selected_country_view_diplo
		window_name = "vassal"
		
		visible = {
		       
		}
		
		effects = {
			vassal_click = {
				if = {
					limit = { NOT = { has_variable = global.open_door } }
					set_variable = { global.open_door = 1 }
				}
				else = {
					clear_variable = global.open_door
				}
			}
		}
        
		triggers = {
		vassal_visible = {
               is_puppet_of = ROOT
            }
		
    }
}
