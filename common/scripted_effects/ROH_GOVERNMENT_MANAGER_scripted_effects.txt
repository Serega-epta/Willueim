#menshikov_system
select_menshikov_system = {
	if = {
		limit = {
			check_variable = {
				selected_government_system = 0
			}
		}
		add_to_variable = { 
			var = reform_points_growth
			value = 1
			tooltip = reform_new_points_growth_tt
		}
		add_political_power = 100
	}
	else = {
		remove_ideas_with_trait = smoilov_system_trait
		remove_ideas_with_trait = naumov_system_trait
	}
	set_variable = {
		var = selected_government_system
		value = 1
		tooltip = select_menshikov_system_tt
	}
	subtract_from_variable = {
		var = reform_points
		value = 20
		tooltip = goverment_system_upgrade_tt
	}
}
materials_science_effect = {
	if = {
		limit = {
			has_idea = materials_science_idea
		}
		custom_effect_tooltip = reform_researched_tt
	}
	else_if = {
		limit = {
			check_variable = {
				selected_government_system = 1
			}
		}
		hidden_effect = {
			add_ideas = materials_science_idea
		}
		subtract_from_variable = {
			var = reform_points

		}

		custom_effect_tooltip = model_is_not_selected_tt
		custom_effect_tooltip = {
			localization_key = goverment_system_upgrade_tt
			RIGHT = 15
		}
	}
