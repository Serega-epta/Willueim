# Эффект обновления нацдуха "Дикарство" (для фокусов Адмиралтейства)

ODS_renewal_national_spirit_ODS_Savagery_effect = {
	if = { 
		limit = { has_idea = ODS_Savagery }
		swap_ideas = {
			remove_idea = ODS_Savagery
			add_idea = ODS_Savagery1
		}
	}
	else_if = { 
		limit = { has_idea = ODS_Savagery1 } 
		swap_ideas = {
			remove_idea = ODS_Savagery1
			add_idea = ODS_Savagery2
		}
	}
	else_if = { 
		limit = { has_idea = ODS_Savagery2 } 
		swap_ideas = {
			remove_idea = ODS_Savagery2
			add_idea = ODS_Savagery3
		}
	}
	else_if = { 
		limit = { has_idea = ODS_Savagery3 } 
		swap_ideas = {
			remove_idea = ODS_Savagery3
			add_idea = ODS_Savagery4
		}
	}
	else_if = { 
		limit = { has_idea = ODS_Savagery4 } 
		swap_ideas = {
			remove_idea = ODS_Savagery4
			add_idea = ODS_Savagery5
		}
	}
	else_if = { 
		limit = { has_idea = ODS_Savagery5 } 
		swap_ideas = {
			remove_idea = ODS_Savagery5
			add_idea = ODS_Savagery6
		}
	}	
}
