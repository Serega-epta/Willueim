start_civil_war = {
	if = {
		limit = {
			has_country_flag = podoliakin_tree
		}
		ROOT = {
			load_focus_tree { 
				tree = army_junta_rodina_unite_podoliakin_tree 
				keep_completed = no 
			}
			retire_country_leader = ARM_michael_onufrenko
		}
		REA = {
			load_focus_tree { 
				tree = army_junta_rodina_unite_onufrenko_tree 
				keep_completed = no 
			}
			promote_character = ARM_michael_onufrenko_rebel
			take_north = yes
		}
	}
	else_if = {
		limit = {
			has_country_flag = onufrenko_tree					
		}
		ROOT = {
			load_focus_tree { 
				tree = army_junta_rodina_unite_onufrenko_tree 
				keep_completed = no 
			}
			promote_character = ARM_michael_onufrenko
			retire_country_leader = ARM_Yuriy_Podolika
		}
		REA = {
			load_focus_tree { 
				tree = army_junta_rodina_unite_podoliakin_tree 
				keep_completed = no 
			}
			promote_character = ARM_Yuriy_Podolika_rebel
			take_southern_suburbs = yes
		}
	}
	#Спавн в Центральной России
	if = {
		limit = {
			check_variable = {
				var = strength_in_central_russia
				value = 75
				compare = greater_than_or_equals 
			}
		}
		ROOT = {
			take_full_central_russia = yes
		}
	}
	else_if = {
		limit = {
			check_variable = {
				var = strength_in_central_russia
				value = 25
				compare = less_than_or_equals 
			}
		}
		REA = {
			take_full_central_russia = yes
		}
	}
	else_if = {
		limit = {
			AND = {
				check_variable = {
					var = strength_in_central_russia
					value = 25
					compare = greater_than
				}
				check_variable = {
					var = strength_in_central_russia
					value = 75
					compare = less_than 
				}
			}
		}
		if = {
			limit = {
				has_country_flag = podoliakin_tree
			}
			REA = {
				take_north_central_russia = yes
			}
			ROOT = {
				take_south_central_russia = yes					
			}
		}
		else_if = {
			limit = {
				has_country_flag = onufrenko_tree
			}
			REA = {
				take_south_central_russia = yes
			}
			ROOT = {
				take_north_central_russia = yes					
			}					
		}
	}
	#Спавн в Приморье
	if = {
		limit = {
			check_variable = {
				var = strength_in_primorye
				value = 75
				compare = greater_than_or_equals 
			}
		}
		ROOT = {
			take_full_primorye = yes
		}
	}
	else_if = {
		limit = {
			check_variable = {
				var = strength_in_primorye
				value = 25
				compare = less_than_or_equals 
			}
		}
		REA = {
			take_full_primorye = yes
		}
	}
	else_if = {
		limit = {
			AND = {
				check_variable = {
					var = strength_in_primorye
					value = 25
					compare = greater_than
				}
				check_variable = {
					var = strength_in_primorye
					value = 75
					compare = less_than 
				}
			}
		}
		if = {
			limit = {
				has_country_flag = podoliakin_tree
			}
			REA = {
				take_north_of_primorye = yes
			}
			ROOT = {
				take_south_of_primorye = yes					
			}
		}
		else_if = {
			limit = {
				has_country_flag = onufrenko_tree
			}
			REA = {
				take_south_of_primorye = yes
			}
			ROOT = {
				take_north_of_primorye = yes					
			}					
		}
	}
	#Спавн на Урале
	if = {
		limit = {
			check_variable = {
				var = strength_in_ural
				value = 75
				compare = greater_than_or_equals 
			}
		}
		ROOT = {
			take_full_ural = yes
		}
	}
	else_if = {
		limit = {
			check_variable = {
				var = strength_in_ural
				value = 25
				compare = less_than_or_equals 
			}
		}
		REA = {
			take_full_ural = yes
		}
	}

		limit = {
			AND = {
				check_variable = {
					var = strength_in_ural
					value = 25
					compare = greater_than
				}
				check_variable = {
					var = strength_in_ural
					value = 75
					compare = less_than 
				}
			}
		}
		if = {
			limit = {
				has_country_flag = podoliakin_tree
			}
			REA = {
				take_north_ural = yes
			}
			ROOT = {
				take_south_ural = yes					
			}
		}
		else_if = {
			limit = {
				has_country_flag = onufrenko_tree
			}
			REA = {
				take_south_ural = yes
			}
			ROOT = {
				take_north_ural = yes					
			}					
		}
	}

	#Спавн дивизий
	if = {
		limit = {
			has_country_flag = podoliakin_tree
		}
		while_loop_effect = {
			limit = {
				check_variable = { count_of_division_Podolyakin > 0 } 
			}
			random_controlled_state = {
				limit = {
					is_fully_controlled_by = ROOT
				}
				create_unit = {
					division = "name = \"lol\" division_template = \"loyalisti\" start_experience_factor = 0.5"
					owner = ROOT
				}	
			}
			subtract_from_variable = {
				var = count_of_division_Podolyakin
				value = 1
			}					
		}

		while_loop_effect = {
			limit = {
				check_variable = { count_of_division_Onufrenko > 0 } 
			}
			REA = {
				random_owned_controlled_state = {
					limit = {
						is_fully_controlled_by = REA
					}
					create_unit = {
						division = "name = \"kek\" division_template = \"loyalisti\" start_experience_factor = 0.5"
						owner = REA
					}	
				}	
			}	
			subtract_from_variable = {
				var = count_of_division_Onufrenko

			}

	if = {
		limit = {
			has_country_flag = onufrenko_tree
		}
		while_loop_effect = {
			limit = {
				check_variable = { count_of_division_Podolyakin > 0 }

				var = count_of_division_Podolyakin
				value = 1
			}					
		}

		while_loop_effect = {
			limit = {
				check_variable = { count_of_division_Onufrenko > 0 } 
			}
			subtract_from_variable = {
				var = count_of_division_Onufrenko
				value = 1
			}

					owner = ROOT
				}	
			}				
		}
	}

	REA = {
		declare_war_on = {
			target = ROOT
			type = annex_everything
		}	
	}	
}
